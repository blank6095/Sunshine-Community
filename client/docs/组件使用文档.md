# 智慧医院挂号预约系统 - 组件使用文档

## 1 组件概述

### 1.1 组件分类

| 分类 | 说明 | 存放目录 |
|------|------|----------|
| 基础UI组件 | 通用UI组件，不包含业务逻辑 | components/base |
| 业务组件 | 包含业务逻辑的复用组件 | components/business |

### 1.2 组件引入方式

```typescript
// 页面中引入基础组件
import CustomButton from '@/components/base/button/index.vue'
import CustomInput from '@/components/base/input/index.vue'

// 页面中引入业务组件
import DoctorCard from '@/components/business/doctor-card/index.vue'

// 使用时需在组件中注册
export default {
  components: {
    CustomButton,
    CustomInput,
    DoctorCard
  }
}
```

## 4 API请求封装

### 4.1 请求工具封装

```typescript
// utils/request.ts
import { getToken, clearToken } from './storage'

const BASE_URL = 'http://localhost:8080/api/v1'

interface ResponseData<T = any> {
  code: number
  message: string
  data: T
}

export const request = {
  async get<T>(url: string, params?: object): Promise<T> {
    const token = getToken()

    const res = await uni.request({
      url: BASE_URL + url,
      method: 'GET',
      data: params,
      header: token ? { Authorization: `Bearer ${token}` } : {}
    })

    const data = res.data as ResponseData<T>

    if (data.code === 401) {
      clearToken()
      uni.navigateTo({ url: '/pages/login/index' })
      throw new Error('登录已过期')
    }

    if (data.code !== 200 && data.code !== 201) {
      uni.showToast({
        title: data.message || '请求失败',
        icon: 'none'
      })
      throw new Error(data.message)
    }

    return data.data
  },

  async post<T>(url: string, data?: object): Promise<T> {
    return this.request({ url, method: 'POST', data })
  },

  async put<T>(url: string, data?: object): Promise<T> {
    return this.request({ url, method: 'PUT', data })
  },

  async delete<T>(url: string): Promise<T> {
    return this.request({ url, method: 'DELETE' })
  },

  async request<T>(options: {
    url: string
    method: string
    data?: object
  }): Promise<T> {
    const token = getToken()

    try {
      const res = await uni.request({
        url: BASE_URL + options.url,
        method: options.method,
        data: options.data,
        header: token ? { Authorization: `Bearer ${token}` } : {}
      })

      const data = res.data as ResponseData<T>

      if (data.code === 401) {
        clearToken()
        uni.navigateTo({ url: '/pages/login/index' })
        throw new Error('登录已过期')
      }

      if (data.code !== 200 && data.code !== 201 && data.code !== 204) {
        uni.showToast({
          title: data.message || '请求失败',
          icon: 'none'
        })
        throw new Error(data.message)
      }

      return data.data
    } catch (error: any) {
      if (error.message === '登录已过期') {
        throw error
      }
      uni.showToast({
        title: error.message || '网络请求失败',
        icon: 'none'
      })
      throw error
    }
  }
}
```

## 5 全局组件注册

在 `main.ts` 中全局注册基础组件：

```typescript
import { createSSRApp } from 'vue'
import App from './App.vue'

// 引入全局组件
import CustomButton from '@/components/base/button/index.vue'
import CustomInput from '@/components/base/input/index.vue'
import CustomCard from '@/components/base/card/index.vue'
import CustomModal from '@/components/base/modal/index.vue'
import CustomSkeleton from '@/components/base/skeleton/index.vue'
import CustomEmpty from '@/components/base/empty/index.vue'
import CustomTabs from '@/components/base/tabs/index.vue'

const app = createSSRApp(App)

// 全局注册组件
app.component('CustomButton', CustomButton)
app.component('CustomInput', CustomInput)
app.component('CustomCard', CustomCard)
app.component('CustomModal', CustomModal)
app.component('CustomSkeleton', CustomSkeleton)
app.component('CustomEmpty', CustomEmpty)
app.component('CustomTabs', CustomTabs)

export function createApp() {
  return {
    app
  }
}
```

全局注册后，页面中无需手动引入即可使用。
